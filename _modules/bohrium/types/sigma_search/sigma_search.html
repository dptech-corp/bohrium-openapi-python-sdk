        **kwargs
    ):
        self.sessions = sessions or []
        self.extra = kwargs



[文档]
    def to_dict(self):
        data = {
            "sessions": [session.to_dict() if isinstance(session, SessionInfo) else session for session in self.sessions]
        }
        if self.extra:
            data.update(self.extra)
        return data
